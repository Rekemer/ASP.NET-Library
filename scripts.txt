CREATE PROCEDURE [dbo].[Book_Get]
    @ISBN BIGINT
AS
BEGIN
    SET NOCOUNT ON;

    SELECT  ISBN,
            Title,
            Author,
            PublishDate,
            Price,
            Publish
    FROM    dbo.Books
    WHERE   ISBN = @ISBN;
END;

CREATE PROCEDURE [dbo].[Book_Insert]
    @ISBN        BIGINT,
    @Title       NVARCHAR(200),
    @Author      NVARCHAR(150),
    @PublishDate DATETIME,
    @Price       DECIMAL(10,2),
    @Publish     BIT
AS
BEGIN
    SET NOCOUNT ON;

    INSERT dbo.Books (ISBN, Title, Author, PublishDate, Price, Publish)
    VALUES (@ISBN, @Title, @Author, @PublishDate, @Price, @Publish);
END;

CREATE PROCEDURE [dbo].[Book_List]
AS
BEGIN
    SET NOCOUNT ON;

    SELECT  ISBN,
            Title,
            Author,
            PublishDate,
            Price,
            Publish
    FROM    dbo.Books
    ORDER BY ISBN;
END;
GO

CREATE PROCEDURE [dbo].[Book_Update]
    @ISBN        BIGINT,
    @Title       NVARCHAR(200),
    @Author      NVARCHAR(150),
    @PublishDate DATETIME,
    @Price       DECIMAL(10,2),
    @Publish     BIT
AS
BEGIN
    SET NOCOUNT ON;

    UPDATE dbo.Books
    SET    Title       = @Title,
           Author      = @Author,
           PublishDate = @PublishDate,
           Price       = @Price,
           Publish     = @Publish
    WHERE  ISBN        = @ISBN;
END;
